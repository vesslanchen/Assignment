\question{}{\textbf{Neutral kaons: oscillations, CP violation and CPT invariance}}\\
Neutral kaon mixing is described by the effective Hamiltonian $H=M-\frac{i}{2}\Gamma$, acting on the $(K^0, \bar{K}^0)$ basis, where $M(\Gamma)$ is the neutral kaon mass (decay) matrix.

\begin{itemize}
    \item [(a)]
\end{itemize}
Consider the following CP-conserving parameterization of the mass matrix as defined in the $(K^0, \bar{K}^0)$ basis:
\begin{align}
M_0=\begin{pmatrix} m & \Delta \\ 
    \Delta & m \end{pmatrix},
\end{align}
where $\Delta$ is real-valued. Determine the mass eigenvalues, $m_\pm$ and basis states ($K_-$, $K_+$) in which $M_0$ becomes diagonal. Using the experimental inputs $m_{K^0}$ and $\Delta m=m_L-m_S$, obtain numerical values for $m$ and $\Delta$.

\begin{itemize}
    \item [(b)]
\end{itemize}
Working in the ($K_-, K_+$) basis, basis, extend the model of (a) to allow for CP
violation by introducing the real-valued parameter $\delta$:
\begin{align}
    M_{\text{CP}}= \begin{pmatrix}
        m_- & -i \delta \\
        i \delta & m_+
     \end{pmatrix},
\end{align}
and assume there is no direct CP violation. This mass matrix corresponds to the
\textit{superweak} model. Assume $\Gamma$ is diagonal in the ($K_-, K_+$) basis with eigenvalues $\Gamma_L$ and $\Gamma_S$. By expressing $M_{\text{CP}}$ in the $(K^0, \bar{K}^0)$ basis, predict the phase, $\varphi_\epsilon$ of $\epsilon$ defined by $|K_S\rangle \propto |K_+\rangle + \epsilon |K_-\rangle$ and $|K_L\rangle \propto |K_-\rangle + \epsilon |K_+\rangle$, and determine $\delta$ from the measured value of $|\epsilon|$.

\begin{itemize}
    \item [(c)]
\end{itemize}
Assume the $\Delta S = \Delta Q$ rule so that, to an excellent approximation, $K^0\to \pi^- l^+ \nu$ and $\bar{K}^0\to \pi^+ l^- \bar{\nu}$, with equal magnitudes of the corresponding decay amplitudes (no direct CP violation in semileptonic decays). Starting from an initially pure $|K^0\rangle$ beam at $t=0$, use the time evolution in terms of $|K_S\rangle$ and $|K_L\rangle$ (including the mixing parameter $\epsilon$ defined in (b)) o derive the time-dependent semileptonic charge asymmetry 
\begin{align}
    A(t)\equiv \frac{N_+(t)-N_-(t)}{N_+(t)+N_-(t)},
\end{align}
where $N_+(t)$ and $N_-(t)$ are the number of semileptonic decays in a small time bin around proper time $t$ that produce $l^+$ and $l^-$, respectively. Express $A(t)$ in terms of $\epsilon$, $\Delta m$ and $\Gamma_{L,S}$ (working to first order in $\epsilon$). Show that at late times (when the beam is $K_L$ dominated), $A(t)\to\delta_L\approx 2\text{Re}(\epsilon)$. Using PDG values for $\Delta m$, $\Gamma_{L,S}$, make a plot of $A(t)$ versus $t/\tau_S$ over the range $0\leq t/\tau_S \leq 10$.

\begin{itemize}
    \item [(d)]

Finally, extend the model in (b) to 
\begin{align}

\begin{align}

\answer{}
\begin{itemize}

\end{itemize}
The determinant condition for the eigenvalues of $M_0$ is
\begin{align}
    \det(M_0 - \lambda \mathbb{1}) = 0 \implies (m-\lambda)^2 - \Delta^2 = 0 \implies \lambda = m \pm \Delta.
\end{align}
Thus, the mass eigenvalues are $m_\pm = m \pm \Delta$. The corresponding eigenvectors are
\begin{align}
    |K_-\rangle = \frac{1}{\sqrt{2}}(|K^0\rangle - |\bar{K}^0\rangle), \\
    |K_+\rangle = \frac{1}{\sqrt{2}}(|K^0\rangle + |\bar{K}^0\rangle).
\end{align}
Using the experimental inputs $m_{K^0} \approx 497.611$ MeV and $\Delta m = m_L - m_S \approx 3.484 \times 10^{-12}$ MeV from PDG, we can identify $m_S = m - \Delta$ and $m_L = m + \Delta$. Thus,
\begin{align}
    m_S = m - \Delta, \\
    m_L = m + \Delta.
\end{align}
Adding these two equations gives $m_S + m_L = 2m \implies m = \frac{m_S + m_L}{2} \approx 497.611$ MeV. And $\Delta = \frac{m_L - m_S}{2} = \frac{\Delta m}{2} \approx 1.742 \times 10^{-12}$ MeV.

\end{itemize}
To express $M_{\text{CP}}$ in the $(K^0, \bar{K}^0)$ basis, we first write the transformation from the $(K_-, K_+)$ basis to the $(K^0, \bar{K}^0)$ basis:
\begin{align}
    \begin{pmatrix}
        |K_-\rangle \\
        |K_+\rangle
    \end{pmatrix} = \frac{1}{\sqrt{2}}\begin{pmatrix}
    1 & -1 \\
    1 & 1
    \end{pmatrix} \begin{pmatrix}
    |K^0\rangle \\
    |\bar{K}^0\rangle
    \end{pmatrix}.
\end{align}
Thus, the transformation matrix is $U = \frac{1}{\sqrt{2}}\begin{pmatrix}
    1 & -1 \\
    1 & 1
\end{pmatrix}$. The mass matrix in the $(K^0, \bar{K}^0)$ basis is then given by
\begin{align}
    M_{\text{CP}}^{(K^0, \bar{K}^0)} = U^\dagger M_{\text{CP}} U.
\end{align}
Calculating this explicitly, we find
\begin{align}
    &M_{\text{CP}}^{(K^0, \bar{K}^0)} = \frac{1}{2}\begin{pmatrix}
    1 & 1 \\
    -1 & 1
    \end{pmatrix} \begin{pmatrix}
    m_- & -i \delta \\
    i \delta & m_+
    \end{pmatrix} \begin{pmatrix}
    1 & -1 \\
    1 & 1
    \end{pmatrix}\\
    =& \frac{1}{2}
    \begin{pmatrix}
        m_-+i\delta & m_+-i\delta \\
        -m_-+i\delta & m_+ +i\delta
    \end{pmatrix} 
    \begin{pmatrix}
    1 & -1 \\
    1 & 1
    \end{pmatrix}\\
    =& \frac{1}{2}
    \begin{pmatrix}
        (m_-+i\delta) + (m_+-i\delta) & -(m_-+i\delta) + (m_+-i\delta) \\
        (-m_-+i\delta) + (m_+ + i\delta) & -(-m_-+i\delta) + (m_+ + i\delta)
    \end{pmatrix}\\
    =& \begin{pmatrix}
    \frac{m_- + m_+}{2} & \frac{m_+ - m_-}{2} - i\delta \\
    \frac{m_+ - m_-}{2} + i\delta & \frac{m_- + m_+}{2}
    \end{pmatrix}\\
    =& \begin{pmatrix}
    m & \Delta - i\delta \\
    \Delta + i\delta & m
    \end{pmatrix}.
\end{align}
Also, we have $\Gamma = \begin{pmatrix}
    \Gamma_S & 0 \\
    0 & \Gamma_L
\end{pmatrix}$ in the $(K_-, K_+)$ basis, which transforms to
\begin{align}
    &\Gamma^{(K^0, \bar{K}^0)} = U^\dagger \Gamma U = \frac{1}{2}\begin{pmatrix}
    1 & 1 \\
    -1 & 1
    \end{pmatrix} \begin{pmatrix}
    \Gamma_S & 0 \\
    0 & \Gamma_L
    \end{pmatrix} \begin{pmatrix}
    1 & -1 \\
    1 & 1
     \end{pmatrix}\\
    =& \frac{1}{2}\begin{pmatrix}
        \Gamma_S & \Gamma_L \\
        -\Gamma_S & \Gamma_L
    \end{pmatrix} 
    \begin{pmatrix}
    1 & -1 \\
    1 & 1
     \end{pmatrix}
    = \begin{pmatrix}
    \frac{\Gamma_S + \Gamma_L}{2} & \frac{\Gamma_L - \Gamma_S}{2} \\
    \frac{\Gamma_L - \Gamma_S}{2} & \frac{\Gamma_S + \Gamma_L}{2}
    \end{pmatrix}\\
    =& \begin{pmatrix}
    \Gamma & \Gamma_{12} \\
    \Gamma_{12} & \Gamma
    \end{pmatrix},
\end{align}
where $\Gamma = \frac{\Gamma_S + \Gamma_L}{2}$ and $\Gamma_{12} = (\Gamma_L - \Gamma_S)/2$. The full Hamiltonian in the $(K^0, \bar{K}^0)$ basis is then 
\begin{align}
    &H = M_{\text{CP}}^{(K^0, \bar{K}^0)} - \frac{i}{2}\Gamma^{(K^0, \bar{K}^0)} = \begin{pmatrix}
    m - \frac{i}{2}\Gamma & \Delta - i\delta - \frac{i}{2}\Gamma_{12} \\
    \Delta + i\delta - \frac{i}{2}\Gamma_{12} & m - \frac{i}{2}\Gamma
    \end{pmatrix}\\
    =& \begin{pmatrix}
        m-\frac{i}{2}\Gamma &\Delta - \frac{i}{2}\Gamma_{12} \\
        \Delta - \frac{i}{2}\Gamma_{12} & m-\frac{i}{2}\Gamma 
    \end{pmatrix}
    + \begin{pmatrix}
        0 & -i\delta \\
        i\delta & 0
    \end{pmatrix}\\
    =& H_0 + H_{\text{CPV}},
\end{align} 
where $H_0$ is the CP-conserving part and $H_{\text{CPV}}$ is the CP-violating part. Now, we can consider perturbation theory to find the effect of $H_{\text{CPV}}$ on the eigenstates of $H_0$. The unperturbed eigenstates of $H_0$ are $|K_+\rangle$ and $|K_-\rangle$ with eigenvalues $m_S - \frac{i}{2}\Gamma_S$ and $m_L - \frac{i}{2}\Gamma_L$, respectively. The perturbed eigenstates are $|K_S\rangle$ and $|K_L\rangle$. Now, we consider the perturbation theory formula for non-degenerate states, given by 
\begin{align}
    |\psi_n^{(1)}\rangle = \sum_{m\neq n} \frac{\langle m|H'|n\rangle}{E_n^{(0)} - E_m^{(0)}} |m\rangle,
\end{align}
where $H'$ is the perturbation and $E_n^{(0)}$ are the unperturbed eigenvalues. In our case, $H' = H_{\text{CPV}}$ and the unperturbed eigenvalues are $E_+ = m_S - \frac{i}{2}\Gamma_S$ and $E_- = m_L - \frac{i}{2}\Gamma_L$. Note that 
\begin{align}
    \langle K_-|H_{\text{CPV}}|K_+\rangle &= (H_{\text{CPV}})_{-+} = (H_{\text{CPV}})_{12} = -i\delta, \\
    \langle K_+|H_{\text{CPV}}|K_-\rangle &= i\delta.
\end{align}
Hence, the first-order correction to the $|K_+\rangle$ state is
\begin{align}
    |K_S\rangle \approx |K_+\rangle + \frac{\langle K_-|H_{\text{CPV}}|K_+\rangle}{E_+ - E_-} |K_-\rangle = |K_+\rangle + \frac{-i\delta}{(m_S - m_L) - \frac{i}{2}(\Gamma_S - \Gamma_L)} |K_-\rangle 
\end{align}
and the first-order correction to the $|K_-\rangle$ state is
\begin{align}
    |K_L\rangle \approx |K_-\rangle + \frac{\langle K_+|H_{\text{CPV}}|K_-\rangle}{E_- - E_+} |K_+\rangle = |K_-\rangle + \frac{i\delta}{(m_L - m_S) - \frac{i}{2}(\Gamma_L - \Gamma_S)} |K_+\rangle.
\end{align}
We can define $\epsilon$, which is the mixing parameter, as
\begin{align}
    &\epsilon = \frac{\langle K_-|H_{\text{CPV}}|K_+\rangle}{E_+ - E_-} = \frac{-i\delta}{(m_S - m_L) - \frac{i}{2}(\Gamma_S - \Gamma_L)}=\frac{-i\delta}{-\Delta m +i \Gamma_{12}}\\
    =& \frac{\delta}{-\Gamma_{12} - i\Delta m} =\frac{\delta}{(\Gamma_S-\Gamma_L) /2 - i(m_L - m_S)}\\
    =& \frac{\delta}{\sqrt{(\Gamma_S-\Gamma_L)^2/4 + (m_L - m_S)^2}} e^{i\varphi_\epsilon},
\end{align}
where $\varphi_\epsilon = \tan^{-1}\left(\frac{2\Delta m}{\Gamma_S - \Gamma_L}\right)$. Using the PDG values for $\Delta m$ and $\Gamma_{L,S}$, we can calculate $\varphi_\epsilon$ and then determine $\delta$ from the measured value of $|\epsilon|$. The PDG values are $m_L-m_S \approx 3.484 \times 10^{-12}$ MeV, $\Gamma_S \approx 7.351 \times 10^{-12}$ MeV, and $\Gamma_L\ll \Gamma_S$. Then 
\begin{align}
    \varphi_\epsilon = \tan^{-1}\left(\frac{2\Delta m}{\Gamma_S - \Gamma_L}\right) \approx \tan^{-1}\left(\frac{2 \times 3.484 \times 10^{-12}}{7.351 \times 10^{-12}}\right) \approx 43.4678^\circ.
\end{align}
For the magnitude of $\epsilon$, we have $|\epsilon| = \frac{\delta}{\sqrt{(\Gamma_S-\Gamma_L)^2/4 + (m_L - m_S)^2}}$. Using the PDG value $|\epsilon| \approx 2.228 \times 10^{-3}$, we can solve for $\delta$:
\begin{align}
    &\delta = |\epsilon| \sqrt{\left(\frac{\Gamma_S - \Gamma_L}{2}\right)^2 + (m_L - m_S)^2} \\
    =&1.128 \times 10^{-14} \text{ MeV}.
\end{align}

where $z$ is a small complex number. Using this approximation, we can write
\begin{align}
    &\varphi_{+-} \approx \varphi_\epsilon + \text{Im}\left(\frac{\epsilon' + \frac{\chi}{2\xi}}{\epsilon}\right), \\
    &\varphi_{00} \approx \varphi_\epsilon + \text{Im}\left(\frac{-2\epsilon' + \frac{\chi}{2\xi}}{\epsilon}\right).
\end{align}
Now, we can derive the relation between the phases:
\begin{align}
    &\frac{2}{3} \varphi_{+-} + \frac{1}{3} \varphi_{00} - \varphi_\epsilon \approx \frac{2}{3} \text{Im}\left(\frac{\epsilon' + \frac{\chi}{2\xi}}{\epsilon}\right) + \frac{1}{3} \text{Im}\left(\frac{-2\epsilon' + \frac{\chi}{2\xi}}{\epsilon}\right)\\
    =& \text{Im}\left(\frac{\chi}{2\xi \epsilon}\right) = \frac{|\chi|}{2|\xi||\epsilon|}\sin(\varphi_\chi - \varphi_\epsilon-\varphi_\xi) \approx \frac{|\chi|}{2|\xi||\epsilon|}\sin\varphi_\epsilon,
\end{align}
where $\varphi_\chi$ is the phase of $\chi$ and $\varphi_\xi$ is the phase of $\xi$. The approximation in the last step assumes that $\varphi_\chi - \varphi_\epsilon - \varphi_\xi \approx \varphi_\epsilon$, which is reasonable if $\chi$ is small and does not introduce a large phase shift. Finally, we can express $|\chi|$ in terms of the mass difference between $K^0$ and $\bar{K}^0$, and $\xi$ in terms of the mass and width differences:
\begin{align}
    |\chi| = |m_{\bar{K}^0} - m_{K^0}|, \quad |\xi| = \sqrt{\left(\frac{\Delta m}{2}\right)^2 + \left(\frac{\Delta \Gamma}{4}\right)^2} \approx \frac{m_L - m_S}{2}.
\end{align} Thus, we have
\begin{align}
    \frac{2}{3} \varphi_{+-} + \frac{1}{3} \varphi_{00} - \varphi_\epsilon \approx \frac{|m_{\bar{K}^0} - m_{K^0}|}{2|\epsilon|(m_L - m_S)}\sin\varphi_\epsilon.
\end{align}
Now for the limit on CPT invariance, we can use the experimental values for $\varphi_{+-}$, $\varphi_{00}$, and $\varphi_\epsilon$ to set an upper limit on $|m_{\bar{K}^0} - m_{K^0}|$. The PDG values are $\varphi_{+-} \approx 43.4^\circ\pm 0.5^\circ$, $\varphi_{00} \approx 43.7^\circ\pm 0.6^\circ$, and $\varphi_\epsilon \approx 43.5^\circ\pm 0.5^\circ$. Using these values, we can calculate the left-hand side of the relation:
\begin{align}
    &\frac{2}{3} \varphi_{+-} + \frac{1}{3} \varphi_{00} - \varphi_\epsilon \approx \frac{2}{3} \times 43.4^\circ + \frac{1}{3} \times 43.7^\circ - 43.5^\circ = 0^\circ \pm \sigma^\circ,
\end{align}
where $\sigma$ is given by the error propagation formula:
\begin{align}
    \sigma = \sqrt{\left(\frac{2}{3}\right)^2 (0.5^\circ)^2 + \left(\frac{1}{3}\right)^2 (0.6^\circ)^2 + (0.5^\circ)^2} \approx 0.63^\circ.
\end{align}
Thus, we have
\begin{align}
    \frac{2}{3} \varphi_{+-} + \frac{1}{3} \varphi_{00} - \varphi_\epsilon = 0^\circ \pm 0.63^\circ.
\end{align}
Now, we can use this result to set an upper limit on $|m_{\bar{K}^0} - m_{K^0}|$:
\begin{align}
    &\left|\frac{2}{3} \varphi_{+-} + \frac{1}{3} \varphi_{00} - \varphi_\epsilon\right| \approx \frac{|m_{\bar{K}^0} - m_{K^0}|}{2|\epsilon|(m_L - m_S)}\sin\varphi_\epsilon   \leq 0.63^\circ \approx 0.011 \text{ radians},  \\
    &|m_{\bar{K}^0} - m_{K^0}| \leq 0.011 \times 2|\epsilon|(m_L - m_S)/\sin\varphi_\epsilon\approx 2.49 \times 10^{-16} \text{ MeV}.
\end{align}
Here we only consider 1$\sigma$ deviation for the limit, which corresponds to a confidence level of about 68\%. For a more stringent limit, we can consider 2$\sigma$ or 3$\sigma$ deviations, which would correspond to confidence levels of about 95\% and 99.7\%, respectively.

\qed
