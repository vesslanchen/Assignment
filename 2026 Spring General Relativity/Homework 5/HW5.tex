\question{1}{}
In the Figure below 

\begin{itemize}
    \item [(a)]In the upper left corner of the figure, there are no solutions where the Universe is currently expanding from a singular state. Derive the analytic form for the boundary of the shaded region.
    \item [(b)] The curve above the line $\Omega_\Lambda = 0$ separates universes which either expand forever or recollapse. Derive the analytic form for this curve.
    \item [(c)] In the lower right, the shaded region corresponds to an exclusion due to the age of the Universe. Assuming $H_0 = 71$ km/s/Mpc calculate the age of the Universe along the boundary of the region.
\end{itemize}


\answer{}
\begin{itemize}
    \item [(a)]
\end{itemize}
We start with the Friedamnn Equation:
\begin{align}
    H^2= H_0^2(\Omega_m a^{-3}+\Omega_k a^{-2} +\Omega_\Lambda),
\end{align}
where $H$ is the Hubble parameter, $H_0$ is the current Hubble parameter, $\Omega_m$ is the matter (dark plus baryonic) density, $\Omega_k$ is the spatial curvature density, $\Omega_\Lambda$ is the vacuum density. If we define $x=1/a$, and try to find the roots for the equation $f(x)=(H/H_0)^2=0$, it's given by:
\begin{align}
    &f(x)=\Omega_m x^3 +\Omega_k x^2 + \Omega_\Lambda  =0\\
    \implies& \Omega_m x^3 + (1-\Omega_m-\Omega_\Lambda)x^2 +\Omega_\Lambda=0,\quad\text{by $\Omega_k=1-\Omega_m-\Omega_\Lambda$}.
\end{align}
If we need the "no big bang" situation, we need $f'(x_0)=0$, where $x_0$ is the root. In other words, $f(x)$ has a double root. Then, we have 
\begin{align}
    f'(x) = 3\Omega_m x^2 + 2(1-\Omega_m-\Omega_\Lambda)x = 0,
\end{align}
and the solution $x_0$ is given by 
\begin{align}
    x_0 = \frac{2(\Omega_m + \Omega_\Lambda - 1)}{3\Omega_m}.
\end{align}
Now we require $f(x_0)=0$:
\begin{align}
    &\Omega_m \left( \frac{2(1-\Omega_m-\Omega_\Lambda)}{3\Omega_m} \right)^3 - (1-\Omega_m-\Omega_\Lambda) \left( \frac{2(1-\Omega_m-\Omega_\Lambda)}{3\Omega_m} \right)^2 + \Omega_\Lambda = 0\\
    \implies &\Omega_m \left( \frac{2\alpha}{3\Omega_m} \right)^3 - \alpha \left( \frac{2\alpha}{3\Omega_m} \right)^2 + \Omega_\Lambda = 0,\quad\text{$\alpha=1-\Omega_m-\Omega_\Lambda$}\\
    \implies&\frac{8\alpha^3}{27\Omega_m^2}-\frac{4\alpha^3}{9\Omega_m^2}=-\Omega_\Lambda,\\
    \implies&\Omega_\Lambda=\frac{4(\Omega_m+\Omega_\Lambda-1)^3}{27 \Omega_m^2}\\
    \implies& 3\Omega_m^{2/3}\Omega_\Lambda^{1/3}=2^{2/3}(\Omega_m+\Omega_\Lambda-1).
\end{align}
We define $\Omega_\Lambda=4\Omega_m \cos^3\theta$. Then we have 
\begin{align}
    &3 \Omega_m \cos\theta = 4\Omega_m\cos^3\theta+\Omega_m -1\\
    \implies& \frac{1-\Omega_m}{\Omega_m}= 4\cos^3\theta-3\cos\theta=\cos3\theta\\
    \implies& \theta =\frac{1}{3}\arccos\left(\frac{1-\Omega_m}{\Omega_m}\right)\\
    \implies& \Omega_\Lambda=4 \Omega_m\left[\cos\left(\frac{1}{3}\arccos\left(\frac{1-\Omega_m}{\Omega_m}\right)\right)\right]^3.
\end{align}
Note that this solution is valid provided $\Omega_m\geq0.5$. If the $\Omega_m\leq0.5$, we can use hyperbolic cosine function to parametrize the equation, and apply the same procedure. Then we can get 
\begin{align}
    \Omega_\Lambda=4 \Omega_m\left[\cosh\left(\frac{1}{3}\text{arccosh}\left(\frac{1-\Omega_m}{\Omega_m}\right)\right)\right]^3.
\end{align}

We start with the same cubic relation derived from the double root condition:
\begin{align}
    \cos(3\theta) = \frac{1-\Omega_m}{\Omega_m}
\end{align}
The general solution for $3\theta$ is given by:
\begin{align}
    3\theta = \arccos\left(\frac{1-\Omega_m}{\Omega_m}\right) + 2n\pi, \quad n = 0, 1, 2
\end{align}
To identify the curve that separates "eternal expansion" from "recollapse," we require the boundary to pass through the specific point $(\Omega_m, \Omega_\Lambda) = (1, 0)$.Testing the roots at $\Omega_m = 1$ (where $\arccos(0) = \pi/2$):
\begin{itemize}
    \item For $n=0$: $\theta = \pi/6$, leading to $\Omega_\Lambda \approx 2.6$ (This corresponds to the No Big Bang boundary).
    \item For $n=1$: $\theta = 5\pi/6$, leading to $\Omega_\Lambda < 0$.
    \item For $n=2$: $\theta = \frac{\pi/2 + 4\pi}{3} = \frac{3\pi}{2}$, which gives $\cos(3\pi/2) = 0$, and thus $\Omega_\Lambda = 0$.
\end{itemize}
Therefore, we choose the branch $n=2$. The analytic form for this curve is:
\begin{align}
    \Omega_\Lambda = 4 \Omega_m \left[ \cos \left( \frac{1}{3} \arccos \left( \frac{1-\Omega_m}{\Omega_m} \right) + \frac{4\pi}{3} \right) \right]^3
\end{align}

\begin{align}

    =&\frac{1}{H_0}\int_0^1 \frac{da}{\sqrt{\Omega_m a^{-1}+\Omega_k +\Omega_\Lambda a^2}}.
\end{align}
We consider the three cases $(\Omega_m,\Omega_\Lambda,\Omega_k)=(2,-1.5,0.5)$, $(3,-0.5,-1.5)$ and $(3,-1.5,-0.5)$, then we get:
\begin{align}
    t_0 = \frac{0.504}{H_0}=6.89 \text{ Gyr},
\end{align}

\begin{align}
    t_0=\frac{0.463}{H_0}=6.33 \text{ Gyr}.
\end{align}
These results are significantly lower than the estimated ages of the oldest stars and globular clusters ($> 12$ Gyr). Thus, the lower-right region of the $\Omega_m - \Omega_\Lambda$ plane is excluded as it leads to a "young universe" paradox.\qed
