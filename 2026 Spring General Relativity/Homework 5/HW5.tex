\question{1}{}
In the Figure below 

\begin{itemize}
    \item [(a)]In the upper left corner of the figure, there are no solutions where the Universe is currently expanding from a singular state. Derive the analytic form for the boundary of the shaded region.
    \item [(b)] The curve above the line $\Omega_\Lambda = 0$ separates universes which either expand forever or recollapse. Derive the analytic form for this curve.
    \item [(c)] In the lower right, the shaded region corresponds to an exclusion due to the age of the Universe. Assuming $H_0 = 71$ km/s/Mpc calculate the age of the Universe along the boundary of the region.
\end{itemize}


\answer{}
\begin{itemize}
    \item [(a)]
\end{itemize}
We start with the Friedamnn Equation:
\begin{align}
    H^2= H_0^2(\Omega_m a^{-3}+\Omega_k a^{-2} +\Omega_\Lambda),
\end{align}
where $H$ is the Hubble parameter, $H_0$ is the current Hubble parameter, $\Omega_m$ is the matter (dark plus baryonic) density, $\Omega_k$ is the spatial curvature density, $\Omega_\Lambda$ is the vacuum density. If we define $x=1/a$, and try to find the roots for the equation $f(x)=(H/H_0)^2=0$, it's given by:
\begin{align}
    &f(x)=\Omega_m x^3 +\Omega_k x^2 + \Omega_\Lambda  =0\\
    \implies& \Omega_m x^3 + (1-\Omega_m-\Omega_\Lambda)x^2 +\Omega_\Lambda=0,\quad\text{by $\Omega_k=1-\Omega_m-\Omega_\Lambda$}.
\end{align}
If we need the "no big bang" situation, we need $f'(x_0)=0$, where $x_0$ is the root. In other words, $f(x)$ has a double root. Then, we have 
\begin{align}
    f'(x) = 3\Omega_m x^2 + 2(1-\Omega_m-\Omega_\Lambda)x = 0,
\end{align}
and the solution $x_0$ is given by 
\begin{align}
    x_0 = \frac{2(\Omega_m + \Omega_\Lambda - 1)}{3\Omega_m}.
\end{align}
Now we require $f(x_0)=0$:
\begin{align}
    &\Omega_m \left( \frac{2(1-\Omega_m-\Omega_\Lambda)}{3\Omega_m} \right)^3 - (1-\Omega_m-\Omega_\Lambda) \left( \frac{2(1-\Omega_m-\Omega_\Lambda)}{3\Omega_m} \right)^2 + \Omega_\Lambda = 0\\
    \implies &\Omega_m \left( \frac{2\alpha}{3\Omega_m} \right)^3 - \alpha \left( \frac{2\alpha}{3\Omega_m} \right)^2 + \Omega_\Lambda = 0,\quad\text{$\alpha=1-\Omega_m-\Omega_\Lambda$}\\
    \implies&\frac{8\alpha^3}{27\Omega_m^2}-\frac{4\alpha^3}{9\Omega_m^2}=-\Omega_\Lambda,\\
    \implies&\Omega_\Lambda=\frac{4(\Omega_m+\Omega_\Lambda-1)^3}{27 \Omega_m^2}\\
    \implies& 3\Omega_m^{2/3}\Omega_\Lambda^{1/3}=2^{2/3}(\Omega_m+\Omega_\Lambda-1).
\end{align}
We define $\Omega_\Lambda=4\Omega_m \cos^3\theta$. Then we have 
\begin{align}
    &3 \Omega_m \cos\theta = 4\Omega_m\cos^3\theta+\Omega_m -1\\
    \implies& \frac{1-\Omega_m}{\Omega_m}= 4\cos^3\theta-3\cos\theta=\cos3\theta\\
    \implies& \theta =\frac{1}{3}\arccos\left(\frac{1-\Omega_m}{\Omega_m}\right)\\
    \implies& \Omega_\Lambda=4 \Omega_m\left[\cos\left(\frac{1}{3}\arccos\left(\frac{1-\Omega_m}{\Omega_m}\right)\right)\right]^3.
\end{align}
Note that this solution is valid provided $\Omega_m\geq0.5$. If the $\Omega_m\leq0.5$, we can use hyperbolic cosine function to parametrize the equation, and apply the same procedure. Then we can get 
\begin{align}
    \Omega_\Lambda=4 \Omega_m\left[\cosh\left(\frac{1}{3}\text{arccosh}\left(\frac{1-\Omega_m}{\Omega_m}\right)\right)\right]^3.
\end{align}

We start with the same cubic relation derived from the double root condition:
\begin{align}
    \cos(3\theta) = \frac{1-\Omega_m}{\Omega_m}
\end{align}
The general solution for $3\theta$ is given by:
\begin{align}
    3\theta = \arccos\left(\frac{1-\Omega_m}{\Omega_m}\right) + 2n\pi, \quad n = 0, 1, 2
\end{align}
To identify the curve that separates "eternal expansion" from "recollapse," we require the boundary to pass through the specific point $(\Omega_m, \Omega_\Lambda) = (1, 0)$.Testing the roots at $\Omega_m = 1$ (where $\arccos(0) = \pi/2$):
\begin{itemize}
    \item For $n=0$: $\theta = \pi/6$, leading to $\Omega_\Lambda \approx 2.6$ (This corresponds to the No Big Bang boundary).
    \item For $n=1$: $\theta = 5\pi/6$, leading to $\Omega_\Lambda < 0$.
    \item For $n=2$: $\theta = \frac{\pi/2 + 4\pi}{3} = \frac{3\pi}{2}$, which gives $\cos(3\pi/2) = 0$, and thus $\Omega_\Lambda = 0$.
\end{itemize}
Therefore, we choose the branch $n=2$. The analytic form for this curve is:
\begin{align}
    \Omega_\Lambda = 4 \Omega_m \left[ \cos \left( \frac{1}{3} \arccos \left( \frac{1-\Omega_m}{\Omega_m} \right) + \frac{4\pi}{3} \right) \right]^3
\end{align}

\question{2}{}
Find the proper distance and determine the time the light was emitted
\begin{itemize}
    \item [(a)] to a galaxy with redshift $z=0.01$
    \item [(b)] to a galaxy with redshift $z=3$
    \item [(c)] What is the angle spanned by the horizon at a redshift $z=1100$? You will need to know the size of the horizon at that redshift and the (angular) distance to that redshift.
\end{itemize}
We will assume matter $(\gamma=1)$ and compare the results for (i) $(\Omega_0,\Omega_\Lambda)=(1,0)$ and (ii) $(\Omega_0,\Omega_\Lambda)=(0.27,0.73)$.

\answer{}
Now, we start with the equations of proper distance and time of emission:
\begin{align}
    d_p &= c \int_0^z \frac{dz}{H},\\
    t_e &= \int_z^{\infty} \frac{dz}{(1+z)H},
\end{align}
where $z$ is red shifting, $H=H_0\sqrt{\Omega_0 a^{-3} +\Omega_\Lambda}$, $a^{-1}=1+z$ Then we can have 
\begin{align}
    d_p =&\frac{c}{H_0} \int_0^z \frac{dz}{\sqrt{\Omega_0(1+z)^3+\Omega_\Lambda}},\\
    t_e =&\frac{1}{H_0} \int_z^{\infty} \frac{dz}{(1+z)\sqrt{\Omega_0(1+z)^3+\Omega_\Lambda}}
\end{align}
\begin{itemize}
    \item [(i)] If $(\Omega_0,\Omega_\Lambda)=(1,0)$,
    \begin{itemize}
    \item [(a)] $(d_p,t_e)=(41.9357 \text{ Mpc},8.97175 \text{ Gyr})$
    \item [(b)] $(d_p,t_e)=(4225 \text{ Mpc}, 1.1383 \text{ Gyr})$
    \end{itemize}
    \item [(ii)] If $(\Omega_0,\Omega_\Lambda)=(0.27,0.73)$
    \begin{itemize}
    \item [(a)] $(d_p,t_e)=(42.1642 \text{ Mpc},13.4245 \text{ Gyr})$
    \item [(b)] $(d_p,t_e)=(6465.85 \text{ Mpc},2.1755 \text{ Gyr})$
    \end{itemize}
\end{itemize}
\begin{itemize}
    \item [(c)]
\end{itemize}
The angular size $\theta$ subtended by the horizon is defined as the ratio of the physical horizon size at redshift $z$ to the angular diameter distance $d_a$:\begin{align}\theta = \frac{d_{H,phys}(z)}{d_a(z)} = \frac{r_H / (1+z)}{d_p / (1+z)} = \frac{r_H}{d_p}.\end{align}Since our models are flat ($\Omega_k = 0$), the factor $(1+z)$ cancels out, and the angle can be directly computed using comoving distances. Now for the angle spanned $\theta$, we have 
\begin{align}
\theta = \frac{r_H(z)}{d_p(z)}
\end{align}
where 
\begin{align}
    r_H = c\int_z^\infty \frac{dz}{H} =\frac{c}{H_0} \int_z^\infty \frac{dz}{\sqrt{\Omega_0(1+z)^3+\Omega_\Lambda}}.
\end{align}
Then we have 
\begin{itemize}
    \item [(i)] If $(\Omega_0,\Omega_\Lambda)=(1,0)$, $\theta=\frac{254.66}{8195}=1.78^\circ$
    \item [(ii)] If $(\Omega_0,\Omega_\Lambda)=(0.27,0.73)$, $\theta=\frac{490.09}{14082.}=1.99^\circ$
\end{itemize}\qed

\question{3}{}
A particle $\chi$ with mass $m=1$ GeV and $g_\chi=2$ as a cross-section for the process $\chi\bar{\chi}\to e^+e^-$ given by
\begin{align}
    \langle \sigma v\rangle =\frac{T^2}{(T^2+m^2)^2}.
\end{align}
Between what temperatures is $\chi$ in thermal equilibrium with the electron? 
\answer{}
The density $n_\chi$ is given by
\begin{align}
    n_\chi = \frac{g_\chi}{(2\pi)^3} \int f(\mathbf{p})d^3p=\frac{g_\chi}{2\pi^2}\int_0^\infty \frac{p^2 dp}{\exp(\sqrt{p^2+m^2}/T)\pm1} 
\end{align}
If we assume the $T\ll m$, we can have 
\begin{align}
    &n_\chi\approx\frac{g_\chi}{2\pi^2}\int_0^\infty \frac{p^2 dp}{\exp(\sqrt{p^2+m^2}/T)}\\
    =&\frac{g_\chi}{2\pi^2}\int_0^\infty p^2 \exp(-\sqrt{p^2+m^2}/T) dp\\
    \approx& \frac{g_\chi}{2\pi^2}\int_0^\infty p^2 \exp{-\left(\frac{m+p^2/2m}{T}\right)}, \quad \text{$E=\sqrt{p^2+m^2}\approx m + \frac{p^2}{2m}$}\\
    =&\frac{g_\chi}{2\pi^2} \sqrt{\frac{\pi}{2}} \left(mT\right)^{3/2} e^{-m/T}\\
    =&g_\chi\left(\frac{mT}{2\pi}\right)^{3/2} e^{-m/T}=2\left(\frac{mT}{2\pi}\right)^{3/2} e^{-m/T}.
\end{align}
Now, we can calculate the interaction rate $\Gamma$, given by 
\begin{align}
    \Gamma= \langle\sigma v \rangle n = \frac{T^2}{(T^2+m^2)^2} 2\left(\frac{mT}{2\pi}\right)^{3/2} e^{-m/T}.
\end{align}
The Hubble parameter is given by 
\begin{align}
    H(T) = \sqrt{\frac{4\pi^3}{45}g} \frac{T^2}{M_{Pl}}\approx1.66\frac{\sqrt{g}T^2}{M_{Pl}},
\end{align}
where the Planck mass $M_{Pl}$ is $1.22\times10^{19}$~GeV. Now we can try to solve the root for the equation $\Gamma(T)=H(T)$, given by 
\begin{align}
    &\frac{T^2}{(T^2+m^2)^2} 2\left(\frac{mT}{2\pi}\right)^{3/2} e^{-m/T} = 1.66\frac{\sqrt{g}T^2}{M_{Pl}}\\
    \implies&\frac{2}{(T^2+1)^2}\left(\frac{T}{2\pi}\right)^{3/2} e^{-1/T} = 1.66\frac{\sqrt{g}}{1.22\times10^{19}},\quad\text{substituting $m = 1$~GeV}
\end{align}
To find the temperature range where $\chi$ is in thermal equilibrium with electrons, we solve $\Gamma(T) \ge H(T)$. Based on the numerical solution of the transcendental equation: 
\begin{align}
    \frac{2}{(T^2 + 1)^2} \left( \frac{T}{2\pi} \right)^{3/2} e^{-1/T} = \frac{1.66 \sqrt{100}}{1.22 \times 10^{19}},
\end{align}
where we simply assume $g\approx100$. Then we can obtain two solutions
\begin{align}
    T_{\text{low}}&=0.0295 \text{ GeV} = 29.5 \text{ MeV},\\
    T_{\text{high}}&=6.13\times10^{6} \text{ GeV}.
\end{align}
Between $29.5$~MeV and $6\times10^{6}$ GeV, the particle $\chi$ can be in thermal equilibrium with the electrons. \qed
